Install pip3:
	sudo apt install python3-pip
Install venv:
	sudo apt-get install python3-venv
	python3 -m venv venv
Activate venv:
    source venv/bin/activate
Install django:
    pip3 install django
Create project:
    django-admin startproject remedial_teaching
Create app:
    python manage.py startapp app_name
Database setup:
    Installation packages:
        sudo apt install python3-dev libmysqlclient-dev default-libmysqlclient-dev
        pip3 install pymysql
        sudo pip3 install mysqlclient
        #sudo apt install mysql-client-core-8.0
        apt-get -s install mariadb-server
    Create database:
        sudo mysql -u root
        CREATE DATABASE database_name; # ex: remedial_teaching
        CREATE USER 'user'@'localhost' IDENTIFIED WITH mysql_native_password BY 'password'; # ex: User: remedial_teaching_user, Password: password
        FLUSH PRIVILEGES;
        GRANT ALL ON database_name.* TO 'user'@'localhost';
    Configuration file:
        Create my.cnf file in remedial_teaching/ with content:
            [client]
            database = database_name
            host = localhost
            user = user
            password = password
            default-character-set = utf8
    Restart mysql:
        sudo systemctl daemon-reload
        sudo systemctl restart mysql

For local dev create a devstack.py file in remedial_teaching/remedial_teaching/ with development settings. ex:
    from os.path import abspath, dirname, join
    DEBUG = True
    ALLOWED_HOSTS = ["*"]
    CORS_ORIGIN_ALLOW_ALL = True
    CORS_ORIGIN_WHITELIST = ["*"]
    # google storage config
